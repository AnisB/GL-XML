121 std::string DTDAttribute::createRegex()
122 {
123  std::string regex = "";
124  std::list<CData*>::iterator it = m_datas->begin();
125 
126  if(it != m_datas->end())
127  {
128  // tous les attributs sont #IMPLIED : la regex est de la forme (...)?
129  // on cherche une chaîne de caractère de la forme <attribut"valeur">
130  // la regex est donc de la forme (attribut"[^"])?, <valeur> pouvant prendre n'importe quel
131  // type de caractère hormis "
132  regex+='(' + (*it)->getContent() + "\"[^\"]*\"" + ")?";
133  it++;
134  }
135  for( ; it != m_datas->end(); it++ )
136  {
137  // on rajoute un espace facultatif (on n'est pas garanti d'avoir un attribut avant
138  // car ils sont facultatifs) avant chacune des attributs
139  regex+= "(\\s?"+ (*it)->getContent() + "\"[^\"]*\"" + ")?";
140  }
141 
142  return regex;
143 }
